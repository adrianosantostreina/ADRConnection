## ‚ö°Ô∏è Quickstart
```delphi
// Create a Connection
uses
  ADRConn.Model.Interfaces;

var
  FConnection: IADRConnection;
begin
  FConnection := CreateConnection;
  FConnection.Params
    .Driver(adrPostgres)
    .Database('demoadrconnection')
    .Server('127.0.0.1')
    .Port(5432)
    .UserName('postgres')
    .Password('postgres');

  FConnection.Connect;    
end.

// Create a Query
var
  FQuery: IADRQuery;
begin
  FQuery := CreateQuery(FConnection);
  FQuery.SQL('select id, name, document, phone')
    .SQL('from person')
    .Open;
end;

```
